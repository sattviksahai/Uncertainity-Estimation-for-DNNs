class Dataset():
    def __init__(self, length_range, frequency_range, amplitude_range, dataset_length):
        self.length_range = length_range
        self.frequency_range = frequency_range
        self.amplitude_range = amplitude_range
        self.dataset_length = dataset_length
        if len(self.length_range) == 2 and len(self.frequency_range) == 2 and len(amplitude_range) == 2:
            print('Inputs are good')
        else:
            sys.exit('invalid range input')

    def __len__(self):
        return self.dataset_length
    
    def sine_generator(self, length, freq, ampl):
        x = np.arange(0,2*length, 0.1)
        sine_wave = ampl*np.sin(freq*x)
        return sine_wave

    def __getitem__(self, idx):
        #seq_len = random.randint(self.length_range[0],self.length_range[1])
        seq_len = sequence_length
        frequency_scalar = random.uniform(self.frequency_range[0],self.frequency_range[1])
        amplitude_scalar = random.uniform(self.amplitude_range[0],self.amplitude_range[1])
        data = self.sine_generator(seq_len, frequency_scalar, amplitude_scalar)
        return np.asarray(data[:int(seq_len/0.1)]), np.asarray(data[int(seq_len/0.1):])

def seq2seq_model():
    # Define Model
    model = Sequential()
    model.add(LSTM(40, return_sequences=True, input_shape=(int(sequence_length/0.1),1)))
    model.add(LSTM(40, return_sequences=False))

    model.add(Dense(80, activation='tanh'))
    model.add(RepeatVector(int(sequence_length/0.1)))

    model.add(LSTM(40, return_sequences=True))
    model.add(LSTM(40, return_sequences=True))

    model.add(TimeDistributed(Dense(2)))
    model.add(Activation('linear'))
    return model

def gaussian_nll(y_true, y_pred):
    """Keras implmementation og multivariate Gaussian negative loglikelihood loss function. 
    This implementation implies diagonal covariance matrix.
    
    Parameters
    ----------
    ytrue: tf.tensor of shape [n_samples, n_dims]
        ground truth values
    ypreds: tf.tensor of shape [n_samples, n_dims*2]
        predicted mu and logsigma values (e.g. by your neural network)
        
    Returns
    -------
    neg_log_likelihood: float
        negative loglikelihood averaged over samples
        
    This loss can then be used as a target loss for any keras model, e.g.:
        model.compile(loss=gaussian_nll, optimizer='Adam') 
    
    """
    mu = y_pred[:,:,0]
    logsigma = y_pred[:,:,1]
    
    mse = -0.5*K.sum(K.square((y_true[:,:,0]-mu)/K.exp(logsigma)),axis=1)
    sigma_trace = -K.sum(logsigma, axis=1)
    log2pi = -0.5*100.0*np.log(2*np.pi)
    
    log_likelihood = mse+sigma_trace+log2pi

    return K.mean(-log_likelihood)

def laplacial_likelihood(y_true, y_pred):
    """ Negative log likelihood. """
    #print(y_pred[:,:,0].shape, y_true.shape)
    likelihood = K.tf.distributions.Laplace(loc=y_pred[:,:,0], scale=y_pred[:,:,1])
    
    return - K.sum(likelihood.log_prob(y_true[:,:,0]), axis=-1)
